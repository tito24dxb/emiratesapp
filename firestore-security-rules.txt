rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper Functions
    function isSignedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function isAdmin() {
      return isSignedIn() && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    function isGovernor() {
      return isSignedIn() && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'governor';
    }

    function isAdminOrGovernor() {
      return isAdmin() || isGovernor();
    }

    function hasSubscription(plan) {
      return isSignedIn() && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.subscription == plan;
    }

    function isPremium() {
      return hasSubscription('Pro') || hasSubscription('VIP');
    }

    // Users Collection
    match /users/{userId} {
      allow read: if isSignedIn();
      allow create: if isOwner(userId);
      allow update: if isOwner(userId) || isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // User Profiles (Public Read)
    match /userProfiles/{profileId} {
      allow read: if true;
      allow write: if isOwner(resource.data.userId) || isAdminOrGovernor();
    }

    // Messages Collection
    match /messages/{messageId} {
      allow read: if isSignedIn() && (
        resource.data.senderId == request.auth.uid ||
        resource.data.recipientId == request.auth.uid ||
        exists(/databases/$(database)/documents/conversations/$(resource.data.conversationId)/members/$(request.auth.uid))
      );
      allow create: if isSignedIn() && request.resource.data.senderId == request.auth.uid;
      allow update: if isSignedIn() && resource.data.senderId == request.auth.uid;
      allow delete: if isOwner(resource.data.senderId) || isAdminOrGovernor();
    }

    // Enhanced Messages Collection
    match /enhancedMessages/{messageId} {
      allow read: if isSignedIn() && (
        resource.data.senderId == request.auth.uid ||
        exists(/databases/$(database)/documents/conversations/$(resource.data.conversationId)/members/$(request.auth.uid))
      );
      allow create: if isSignedIn() && request.resource.data.senderId == request.auth.uid;
      allow update: if isSignedIn() && (
        resource.data.senderId == request.auth.uid ||
        exists(/databases/$(database)/documents/conversations/$(resource.data.conversationId)/members/$(request.auth.uid))
      );
      allow delete: if isOwner(resource.data.senderId) || isAdminOrGovernor();
    }

    // Conversations Collection
    match /conversations/{conversationId} {
      allow read: if isSignedIn() && (
        request.auth.uid in resource.data.members ||
        isAdminOrGovernor()
      );
      allow create: if isSignedIn() && request.auth.uid in request.resource.data.members;
      allow update: if isSignedIn() && request.auth.uid in resource.data.members;
      allow delete: if isAdminOrGovernor();

      match /members/{memberId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn() && (
          request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.members ||
          isAdminOrGovernor()
        );
      }
    }

    // Notifications Collection
    match /notifications/{notificationId} {
      allow read: if isSignedIn() && resource.data.userId == request.auth.uid;
      allow create: if isAdminOrGovernor();
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId) || isAdminOrGovernor();
    }

    // FCM Notifications Collection
    match /fcmNotifications/{notificationId} {
      allow read: if isSignedIn() && resource.data.userId == request.auth.uid;
      allow create: if isAdminOrGovernor() || isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId) || isAdminOrGovernor();
    }

    // FCM Tokens Collection
    match /fcmTokens/{tokenId} {
      allow read: if isOwner(resource.data.userId) || isAdminOrGovernor();
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId) || isAdminOrGovernor();
    }

    // Typing Indicators Collection
    match /typingIndicators/{indicatorId} {
      allow read: if isSignedIn();
      allow write: if isSignedIn() && request.resource.data.userId == request.auth.uid;
    }

    // User Presence Collection
    match /userPresence/{userId} {
      allow read: if isSignedIn();
      allow write: if isOwner(userId);
    }

    // Courses Collection
    match /courses/{courseId} {
      allow read: if isSignedIn();
      allow create: if isAdminOrGovernor();
      allow update: if isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // Modules Collection
    match /modules/{moduleId} {
      allow read: if isSignedIn();
      allow create: if isAdminOrGovernor();
      allow update: if isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // Main Modules Collection
    match /mainModules/{moduleId} {
      allow read: if isSignedIn();
      allow create: if isAdminOrGovernor();
      allow update: if isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // Enrollments Collection
    match /enrollments/{enrollmentId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if isOwner(resource.data.userId) || isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // Progress Tracking Collection
    match /progress/{progressId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow write: if isOwner(resource.data.userId);
    }

    // Exam Results Collection
    match /examResults/{resultId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // Bug Reports Collection
    match /bugReports/{reportId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn();
      allow update: if isAdminOrGovernor() || isOwner(resource.data.reportedBy);
      allow delete: if isGovernor();
    }

    // Support Chat Collection
    match /supportChats/{chatId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow update: if isOwner(resource.data.userId) || isAdminOrGovernor();
      allow delete: if isGovernor();
    }

    // System Control Collection
    match /systemControl/{controlId} {
      allow read: if isSignedIn();
      allow write: if isGovernor();
    }

    // Feature Flags Collection
    match /featureFlags/{flagId} {
      allow read: if isSignedIn();
      allow write: if isGovernor();
    }

    // Feature Shutdowns Collection
    match /featureShutdowns/{shutdownId} {
      allow read: if isSignedIn();
      allow write: if isGovernor();
    }

    // System Announcements Collection
    match /systemAnnouncements/{announcementId} {
      allow read: if isSignedIn();
      allow write: if isAdminOrGovernor();
    }

    // Leaderboard Collection
    match /leaderboard/{entryId} {
      allow read: if isSignedIn();
      allow write: if isAdminOrGovernor();
    }

    // Rewards Collection
    match /rewards/{rewardId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow write: if isAdminOrGovernor();
    }

    // File Storage Metadata Collection
    match /fileMetadata/{fileId} {
      allow read: if isSignedIn() && (
        resource.data.uploadedBy == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow create: if isSignedIn() && request.resource.data.uploadedBy == request.auth.uid;
      allow update: if isOwner(resource.data.uploadedBy) || isAdminOrGovernor();
      allow delete: if isOwner(resource.data.uploadedBy) || isGovernor();
    }

    // Storage Quota Collection
    match /storageQuota/{userId} {
      allow read: if isOwner(userId) || isAdminOrGovernor();
      allow write: if isAdminOrGovernor();
    }

    // Login Activity Collection
    match /loginActivity/{activityId} {
      allow read: if isSignedIn() && (
        resource.data.userId == request.auth.uid ||
        isAdminOrGovernor()
      );
      allow create: if isOwner(request.resource.data.userId);
      allow update: if isOwner(resource.data.userId);
      allow delete: if isGovernor();
    }

    // Analytics Events Collection (Write-only for users)
    match /analyticsEvents/{eventId} {
      allow read: if isAdminOrGovernor();
      allow create: if isSignedIn();
      allow update: if false;
      allow delete: if isGovernor();
    }

    // Admin Logs Collection
    match /adminLogs/{logId} {
      allow read: if isAdminOrGovernor();
      allow create: if isAdminOrGovernor();
      allow update: if false;
      allow delete: if isGovernor();
    }

    // Default Deny All Other Collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
